// Function to calculate the quality points based on the grade function calculateQualityPoints(grade) { switch (grade) { case 'A+': return 4.0; case 'A': return 4.0; case 'A-': return 3.7; case 'B+': return 3.3; case 'B': return 3.0; case 'B-': return 2.7; case 'C+': return 2.3; case 'C': return 2.0; case 'C-': return 1.7; case 'D+': return 1.3; case 'D': return 1.0; case 'D-': return 0.7; case 'F': return 0.0; default: return 0.0; } } // Function to calculate GPA function calculateGPA() { var totalQualityPoints = 0; var totalCredits = 0; var inputGroups = document.querySelectorAll('.input-group'); inputGroups.forEach(function(group) { var courseName = group.querySelector('input[type="text"]').value; var credits = parseFloat(group.querySelector('input[type="number"]').value); var grade = group.querySelector('select').value; if (courseName && credits && grade) { var qualityPoints = calculateQualityPoints(grade); totalQualityPoints += qualityPoints * credits; totalCredits += credits; } }); var gpa = totalQualityPoints / totalCredits; document.getElementById('result').textContent = "GPA: " + gpa.toFixed(2); } // Function to add a new row function addRow() { var container = document.getElementById('container'); var newInputGroup = document.createElement('div'); newInputGroup.classList.add('input-group'); newInputGroup.innerHTML = ` `; container.appendChild(newInputGroup); } // Event listeners document.getElementById('add-row').addEventListener('click', addRow); document.getElementById('calculate').addEventListener('click', calculateGPA);
